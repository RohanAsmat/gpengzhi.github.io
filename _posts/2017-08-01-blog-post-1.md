## 78. Subsets
<pre>
class Solution(object):
    
    def subsets(self, nums):
        """
        :type nums: List[int]
        :rtype: List[List[int]]
        """
        res = []
        self.helper(nums, [], res, len(nums))
        return res
        
    def helper(self, nums, path, res, n):
        if len(path) <= n:
            res.append(path)
        for i in range(len(nums)):
            self.helper(nums[i+1:], path+[nums[i]], res, n)
</pre>
Time Complexity: O(# of combinations), Space Complexity: O(n*(# of combinations)).

## 137. Single Number II
<pre>
class Solution(object):
    
    def singleNumber(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        res = 32*[0]
        flag = 0
        for num in nums:
            if num < 0:
                flag += 1
                num = -num
            i = 31
            while num != 0:
                if num%2 == 1:
                    res[i] += 1
                num = num>>1
                i -= 1 
        for i in range(32):
            res[i] = res[i]%3
        resnum = 0
        for i in range(31,-1,-1):
            if res[i] == 1:
                resnum = resnum + 2**(31-i)
        if flag%3 == 1:
            return -resnum
        else:
            return resnum
</pre>
Time Complexity: O(n), Space Complexity: O(1).

## 169. Majority Element
<pre>
class Solution(object):
    
    def majorityElement(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        dict = {}
        n = len(nums)
        for i in range(n):
            if nums[i] in dict:
                dict[nums[i]] += 1
            else:
                dict[nums[i]] = 1
            if dict[nums[i]] > n/2:
                return nums[i]
</pre>
Time Complexity: O(n), Space Complexity: O(n).

## 187. Repeated DNA Sequences
<pre>
class Solution(object):
    
    def findRepeatedDnaSequences(self, s):
        """
        :type s: str
        :rtype: List[str]
        """
        seen = set()
        repeat = set()
        for i in range(len(s)-9):
            substring = s[i:i+10]
            if substring in seen:
                repeat.add(substring)
            else:
                seen.add(substring)
        return list(repeat)
</pre>

class Solution:

## 201. Bitwise AND of Numbers Range
<pre>
class Solution(object):
    
    def rangeBitwiseAnd(self, m, n):
        """
        :type m: int
        :type n: int
        :rtype: int
        """
        count = 0
        while m != n:
            m = m>>1
            n = n>>1
            count += 1
        return m << count
</pre>
Time Complexity: O(log(n)), Space Complexity: O(1).

## 260. Single Number III
<pre>
class Solution(object):
    
    def singleNumber(self, nums):
        """
        :type nums: List[int]
        :rtype: List[int]
        """
        res = 0
        for num in nums:
            res = res^num
        num_tp = 1
        while res%num_tp == 0:
            num_tp = 2*num_tp
        res1 = 0
        res2 = 0
        for num in nums:
            if num & (num_tp/2) == (num_tp/2):
                res1 = res1^num
            else:
                res2 = res2^num
        return [res1, res2]
</pre>
Time Complexity: O(n), Space Complexity: O(1).

## 318. Maximum Product of Word Lengths
<pre>
class Solution(object):
    
    def maxProduct(self, words):
        """
        :type words: List[str]
        :rtype: int
        """
        bitset = []
        for word in words:
            bit_tp = 0
            for w in word:
                bit_tp = bit_tp | 1 << (ord(w)-ord('a'))
            bitset.append(bit_tp)
        res = 0
        for i in range(len(bitset)):
            for j in range(i+1,len(bitset)):
                if bitset[i] & bitset[j] == 0 and len(words[i])*len(words[j]) > res:
                    res = len(words[i])*len(words[j])
        return res
</pre>
Time Complexity: O(n^2), Space Complexity: O(n).

## 338. Counting Bits
Very interesting.
<pre>
class Solution(object):
    
    def countBits(self, num):
        """
        :type num: int
        :rtype: List[int]
        """
        res = (num+1)*[0]
        offset = 1
        for index in range(1,num+1):
            if 2*offset == index:
                offset = 2*offset
            res[index] = res[index-offset]+1
        return res
</pre>
Time Complexity: O(n), Space Complexity: O(1).

## 389. Find the Difference

class Solution(object):
    
    def findTheDifference(self, s, t):
        """
        :type s: str
        :type t: str
        :rtype: str
        """
        c = s+t
        xor = 0
        for i in range(len(c)):
            xor = xor^ord(c[i])
        return chr(xor)
</pre>
Time Complexity: O(n), Space Complexity: O(n).

## 397. Integer Replacement
<pre>
class Solution(object):
    
    def __init__(self):
        self.dict = {1:0}
    
    def integerReplacement(self, n):
        """
        :type n: int
        :rtype: int
        """
        if n in self.dict:
            return self.dict[n]
        if n%2 == 0:
            self.dict[n] = self.integerReplacement(n/2)+1
        else:
            self.dict[n] = min(self.integerReplacement(n+1), self.integerReplacement(n-1))+1 
        return self.dict[n]
</pre>
Time Complexity: O(n), Space Complexity: O(n).

## 401. Binary Watch
<pre>
class Solution(object):
    
    def readBinaryWatch(self, num):
        """
        :type num: int
        :rtype: List[str]
        """
        res = []
        for h in range(12):
            for m in range(60):
                if bin(h).count('1') + bin(m).count('1') == num:
                    if m < 10:
                        res.append(str(h)+':0'+str(m))
                    else:
                        res.append(str(h)+':'+str(m))
        return res
</pre>
Time Complexity: O(1), Space Complexity: O(1).

        """
        :type x: int
        :type y: int

## 477. Total Hamming Distance
<pre>
class Solution(object):
    
    def totalHammingDistance(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        p = 32*[0]
        s = len(nums)
        for num in nums:
            i = 0
            while num != 0:
                if num%2 == 1:
                    p[i] += 1
                i += 1
                num = num/2
        res = 0
        for i in range(len(p)):
            res = res + (s-p[i])*p[i]
        return res
</pre>
Time Complexity: O(n), Space Complexity: O(1).
