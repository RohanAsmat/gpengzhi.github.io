## 1. Two Sum

class Solution(object):
    
    def twoSum(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: List[int]
        """
        if len(nums) <= 1:
            return False
        dict = {}
        for i in range(len(nums)):
            if nums[i] in dict:
                return [dict[nums[i]],i]
            else:
                dict[target-nums[i]] = i
</pre>
Time Complexity: O(n), Space Complexity: O(n).

## 380. Insert Delete GetRandom O(1)
<pre>
class RandomizedSet(object):

    def __init__(self):
        """
        Initialize your data structure here.
        """
        self.nums = []
        self.pos = {}

    def insert(self, val):
        """
        Inserts a value to the set. Returns true if the set did not already contain the specified element.
        :type val: int
        :rtype: bool
        """
        if val not in self.pos:
            self.nums.append(val)
            self.pos[val] = len(self.nums)-1
            return True
        else:
            return False

    def remove(self, val):
        """
        Removes a value from the set. Returns true if the set contained the specified element.
        :type val: int
        :rtype: bool
        """
        if val in self.pos:
            idx = self.pos[val]
            last = self.nums[-1]
            self.nums[idx] = self.nums[-1]
            self.pos[last] = idx
            self.nums.pop()
            self.pos.pop(val)
            return True
        else:
            return False 

    def getRandom(self):
        """
        Get a random element from the set.
        :rtype: int
        """
        return self.nums[random.randint(0, len(self.nums)-1)]
</pre>
Time Complexity: O(1), Space Complexity: O(n).

## 389. Find the Difference
<pre>
class Solution(object):
    
    def findTheDifference(self, s, t):
        """
        :type s: str
        :type t: str
        :rtype: str
        """
        c = s+t
        xor = 0
        for i in range(len(c)):
            xor = xor^ord(c[i])
        return chr(xor)
</pre>
Time Complexity: O(n), Space Complexity: O(1).

## 409. Longest Palindrome
<pre>
class Solution(object):
    
    def longestPalindrome(self, s):
        """
        :type s: str
        :rtype: int
        """
        dict = {}
        for ss in s:
            if ss in dict:
                dict[ss] += 1
            else:
                dict[ss] = 1
        res = 0
        for key in dict.keys():
            res += dict[key]/2*2
            dict[key] = dict[key]%2
        if sum(dict.values()) == 0:
            return res
        else:
            return res+1
</pre>
Time Complexity: O(n), Space Complexity: O(n).

## 438. Find All Anagrams in a String
<pre>
class Solution(object):
    
    def findAnagrams(self, s, p):
        """
        :type s: str
        :type p: str
        :rtype: List[int]
        """
        dict_p = {}
        for p_s in p:
            if p_s in dict_p:
                dict_p[p_s] += 1
            else:
                dict_p[p_s] = 1
        res = []
        dict_s = {}
        for s_s in s[0:len(p)]:
            if s_s in dict_s:
                dict_s[s_s] += 1
            else:
                dict_s[s_s] = 1
        if dict_s == dict_p:
            res.append(0)
        
        for i in range(len(p),len(s)):
            if s[i] in dict_s:
                dict_s[s[i]] += 1
            else:
                dict_s[s[i]] = 1
            dict_s[s[i-len(p)]] -= 1
            if dict_s[s[i-len(p)]] == 0:
                del dict_s[s[i-len(p)]]
            if dict_s == dict_p:
                res.append(i-len(p)+1)
        return res
</pre>
Time Complexity: O(len(s)), Space Complexity: O(len(p)).

## 447. Number of Boomerangs
<pre>
class Solution(object):
    
    def numberOfBoomerangs(self, points):
        """
        :type points: List[List[int]]
        :rtype: int
        """
        res = 0
        for p in points:
            dict = {}
            for q in points:
                x = p[0]-q[0]
                y = p[1]-q[1]
                if x**2+y**2 in dict:
                    dict[x**2+y**2] += 1
                else:
                    dict[x**2+y**2] = 1
            for key in dict.keys():
                res += dict[key]*(dict[key]-1)
        return res
</pre>
Time Complexity: O(n^2), Space Complexity: O(n).

## 451. Sort Characters By Frequency
<pre>
class Solution(object):
    
    def frequencySort(self, s):
        """
        :type s: str
        :rtype: str
        """
        dict1 = {}
        for char in s:
            if char in dict1:
                dict1[char] += 1
            else:
                dict1[char] = 1
        dict2 = {}
        for key in dict1.keys():
            if dict1[key] in dict2:
                dict2[dict1[key]].append(key)
            else:
                dict2[dict1[key]] = [key]
        res = ''
        for key in sorted(dict2.keys(), reverse=True):
            for i in range(len(dict2[key])):
                res += key*dict2[key][i]
        return res
</pre>
Time Complexity: O(nlog(n)), Space Complexity: O(n).

## 454. 4Sum II
<pre>
class Solution(object):
    
    def fourSumCount(self, A, B, C, D):
        """
        :type A: List[int]
        :type B: List[int]
        :type C: List[int]
        :type D: List[int]
        :rtype: int
        """
        dict = {}
        for i in range(len(A)):
            for j in range(len(B)):
                if A[i]+B[j] in dict:
                    dict[A[i]+B[j]] += 1
                else:
                    dict[A[i]+B[j]] = 1
        res = 0
        for k in range(len(C)):
            for l in range(len(D)):
                if -(C[k]+D[l]) in dict:
                    res += dict[-(C[k]+D[l])]
        return res
</pre>
Time Complexity: O(n^2), Space Complexity: O(n^2).

## 463. Island Perimeter
<pre>
class Solution(object):
    
    def islandPerimeter(self, grid):
        """
        :type grid: List[List[int]]
        :rtype: int
        """
        res = 0
        for i in range(len(grid)):
            for j in range(len(grid[0])):
                if grid[i][j] == 1:
                    res += 4
                    if j-1 >= 0 and grid[i][j-1] == 1:
                        res -= 1
                    if j+1 <= len(grid[0])-1 and grid[i][j+1] == 1:
                        res -= 1
                    if i-1 >= 0 and grid[i-1][j] == 1:
                        res -= 1
                    if i+1 <= len(grid)-1 and grid[i+1][j] == 1:
                        res -= 1
        return res
</pre>
Time Complexity: O(n^2), Space Complexity: O(1).

## 508. Most Frequent Subtree Sum
<pre>
class Solution(object):
    
    def __init__(self):
        self.dict = {}
    
    def findFrequentTreeSum(self, root):
        """
        :type root: TreeNode
        :rtype: List[int]
        """
        if root != None:
            self.allSumTree(root)
            res = []
            frequency = max(self.dict.values())
            for key in self.dict.keys():
                if self.dict[key] == frequency:
                    res.append(key)
            return res
        else:
            return []
        
    def allSumTree(self, root):
        if root != None:
            self.allSumTree(root.left)
            if self.sumSubtree(root) in self.dict:
                self.dict[self.sumSubtree(root)] += 1
            else:
                self.dict[self.sumSubtree(root)] = 1
            self.allSumTree(root.right)
        
    def sumSubtree(self, root):
        if root == None:
            return 0
        else:
            return root.val + self.sumSubtree(root.left) + self.sumSubtree(root.right)
</pre>
Time Complexity: O(n^2), Space Complexity: O(n^2).

class Solution(object):
    
    def findMaxLength(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """             
        res = 0
        sum = 0
        dict = {0:[-1]}

            else:
                sum -= 1

## 535. Encode and Decode TinyURL
<pre>
class Codec:
    
    def __init__(self):
        self.cache = []

    def encode(self, longUrl):
        """Encodes a URL to a shortened URL.
        :type longUrl: str
        :rtype: str
        """
        self.cache.append(longUrl)
        return 'http://tinyurl.com/'+str(len(self.cache)-1)

    def decode(self, shortUrl):
        """Decodes a shortened URL to its original URL.
        :type shortUrl: str
        :rtype: str
        """
        shorturl = shortUrl.split('/')
        return self.cache[int(shorturl[-1])]
</pre>

## 554. Brick Wall
<pre>
class Solution(object):
    
    def leastBricks(self, wall):
        """
        :type wall: List[List[int]]
        :rtype: int
        """
        m = len(wall)
        dict = {}
        for i in range(m):
            temp = 0
            for j in range(len(wall[i])-1):
                temp += wall[i][j]
                if temp in dict:
                    dict[temp] += 1
                else:
                    dict[temp] = 1
        if len(dict) == 0:
            return m
        else:
            return m - max(dict.values())
</pre>
Time Complexity: O(mn), Space Complexity: O(mn).

## 594. Longest Harmonious Subsequence
<pre>
class Solution(object):
    
    def findLHS(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        dict = {}
        for num in nums:
            if num in dict:
                dict[num] += 1
            else:
                dict[num] = 1
        res = 0
        for key in dict.keys():
            if key+1 in dict:
                res = max(res, dict[key]+dict[key+1])
            if key-1 in dict:
                res = max(res, dict[key]+dict[key-1])
        return res
</pre>
Time Complexity: O(n), Space Complexity: O(n).

## 599. Minimum Index Sum of Two Lists

class Solution(object):
    
    def findRestaurant(self, list1, list2):
        """
        :type list1: List[str]
        :type list2: List[str]
        :rtype: List[str]
        """
        dict1 = {}
        for i in range(len(list1)):
            dict1[list1[i]] = i
        dict2 = {}
        for i in range(len(list2)):
            dict2[list2[i]] = i
        res = []
        rak = []
        for key in dict1.keys():
            if key in dict2:
                res.append(key)
                rak.append(dict1[key]+dict2[key])
        result = []
        for i in range(len(res)):
            if rak[i] == min(rak):
                result.append(res[i])
        return result
</pre>
